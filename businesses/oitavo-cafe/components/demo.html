Usage

How to use components in your Next.js project

import { Button, Card, CardHeader, CardContent, Input, Alert } from '@/components';

export function ContactForm() {
  return (
    <Card variant="featured">
      <CardHeader>
        <CardTitle>Entre em contato</CardTitle>
        <CardDescription>Vamos conversar sobre seu projeto</CardDescription>
      </CardHeader>
      <CardContent className="space-y-6">
        <Input label="Nome completo" placeholder="Seu nome" />
        <Input label="Email" type="email" placeholder="seu@email.com" />
        <Textarea label="Mensagem" placeholder="Conte-nos sobre seu projeto..." />
        <Button variant="primary">Quero agendar</Button>
      </CardContent>
    </Card>
  );
}
