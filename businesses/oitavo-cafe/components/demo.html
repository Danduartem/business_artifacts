function MetricCardWithLoading() {
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const timer = setTimeout(() => setLoading(false), 2000);
    return () => clearTimeout(timer);
  }, []);

  return (
    <div className={`skeleton-wrapper ${loading ? 'loading' : 'loaded'}`}>
      {/* Actual content - hidden while loading */}
      <div className="skeleton-content">
        <MetricCard data={data} />

      {loading && (
        <div className="skeleton-placeholder">

              <div className="skeleton-card-title"></div>
              <div className="skeleton-card-subtitle"></div>

// Accessible Trend Indicator
function TrendIndicator({ value, previousValue }) {
  const change = ((value - previousValue) / previousValue) * 100;
  const isPositive = change > 0;
  const isNegative = change < 0;

  return (
    <span className="trend-accessible">
      {/* Visual arrow - hidden from screen readers */}
      <span aria-hidden="true">
        {isPositive ? '↑' : isNegative ? '↓' : '→'}
      </span>

      {/* Screen reader context */}
      <span className="sr-only">
        {isPositive ? 'Aumento de' : isNegative ? 'Redução de' : 'Sem alteração:'}
      </span>

      {Math.abs(change).toFixed(1)}%

      <span className="sr-only">
        em relação ao período anterior
      </span>
    </span>
  );
}

// Accessible Chart with Data Table
function AccessibleBarChart({ data, title }) {
  return (
    <div
      className="chart-accessible"
      role="img"
      aria-label={`Gráfico: ${title}. ${generateSummary(data)}`}
    >
      {/* Visual chart */}
      <div aria-hidden="true">
        <BarChartVisual data={data} />
      </div>

      {/* Hidden data table for screen readers */}
      <table className="chart-data-table">
        <caption>{title}</caption>
        <thead>
          <tr>
            <th scope="col">Categoria</th>
            <th scope="col">Valor</th>
          </tr>
        </thead>
        <tbody>
          {data.map(item => (
            <tr key={item.label}>
              <td>{item.label}</td>
              <td>{item.value}</td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}

// Custom dropdown component
interface SelectOption {
  value: string;
  label: string;
  disabled?: boolean;
}

interface SelectProps {
  options: SelectOption[];
  value?: string;
  onChange: (value: string) => void;
  placeholder?: string;
  label?: string;
  searchable?: boolean;
  multi?: boolean;
}

export function Select({
  options, value, onChange, placeholder, label, searchable, multi
}: SelectProps) {
  const [isOpen, setIsOpen] = useState(false);
  const [search, setSearch] = useState('');

  const filtered = searchable
    ? options.filter(o => o.label.toLowerCase().includes(search.toLowerCase()))
    : options;

  return (
    <div className="select-custom">
      {label && <label className="block text-sm font-medium text-gray-700 mb-2">{label}</label>}
      <button
        type="button"
        className="select-custom-trigger"
        aria-haspopup="listbox"
        aria-expanded={isOpen}
        onClick={() => setIsOpen(!isOpen)}
      >
        <span className={!value ? 'select-placeholder' : ''}>
          {value ? options.find(o => o.value === value)?.label : placeholder}
        </span>
        <ChevronDownIcon className="select-chevron w-4 h-4" />
      </button>
      <div className={`select-menu ${isOpen ? 'is-open' : ''}`} role="listbox">
        {searchable && (
          <div className="select-search">
            <input
              type="text"
              placeholder="Buscar..."
              value={search}
              onChange={(e) => setSearch(e.target.value)}
            />
          </div>
        )}
        {filtered.map(option => (
          <div
            key={option.value}
            className="select-option"
            role="option"
            aria-selected={option.value === value}
            aria-disabled={option.disabled}
            tabIndex={0}
            onClick={() => {
              onChange(option.value);
              setIsOpen(false);
            }}
          >
            {option.label}
          </div>
        ))}
      </div>
    </div>
  );
}

interface TabsProps {
  tabs: { id: string; label: string; badge?: number; disabled?: boolean }[];
  activeTab: string;
  onChange: (id: string) => void;
  variant?: 'underline' | 'pills' | 'compact';
}

export function Tabs({ tabs, activeTab, onChange, variant = 'underline' }: TabsProps) {
  const listClass = variant === 'pills' ? 'tabs-list tabs-list-pills'
    : variant === 'compact' ? 'tabs-list tabs-list-compact' : 'tabs-list';

  return (
    <div className="tabs">
      <div className={listClass} role="tablist">
        {tabs.map(tab => (
          <button
            key={tab.id}
            className={`tab-trigger ${activeTab === tab.id ? 'is-active' : ''}`}
            role="tab"
            aria-selected={activeTab === tab.id}
            disabled={tab.disabled}
            onClick={() => onChange(tab.id)}
          >
            {tab.label}
            {tab.badge !== undefined && <span className="tab-badge">{tab.badge}</span>}
          </button>
        ))}
      </div>
    </div>
  );
}

interface ToggleProps {
  checked: boolean;
  onChange: (checked: boolean) => void;
  label?: string;
  disabled?: boolean;
  size?: 'sm' | 'md' | 'lg';
}

export function Toggle({
  checked, onChange, label, disabled, size = 'md'
}: ToggleProps) {
  const sizeClass = size === 'sm' ? 'toggle-sm' : size === 'lg' ? 'toggle-lg' : '';

  return (
    <label className={`toggle ${sizeClass}`}>
      <input
        type="checkbox"
        className="toggle-input"
        checked={checked}
        disabled={disabled}
        onChange={(e) => onChange(e.target.checked)}
      />
      <div className="toggle-track">
        <div className="toggle-thumb" />
      </div>
      {label && <span className="toggle-label">{label}</span>}
    </label>
  );
}

// Settings group pattern

      </div>
      <Toggle checked={checked} onChange={onChange} />

}

// Full error state component
interface ErrorStateProps {
  title: string;
  message: string;
  code?: string;
  onRetry?: () => void;
  variant?: 'default' | 'network' | 'timeout';
}

export function ErrorState({
  title, message, code, onRetry, variant = 'default'
}: ErrorStateProps) {
  const variantClass = variant === 'network' ? 'error-network'
    : variant === 'timeout' ? 'error-timeout' : '';

  return (
    <div className={`error-state ${variantClass}`}>
      <AlertTriangleIcon className="error-state-icon" />
      <h4 className="error-state-title">{title}</h4>
      <p className="error-state-message">{message}</p>
      {code && <span className="error-state-code">{code}</span>}
      {onRetry && (
        <div className="error-state-actions">
          <button className="error-btn-retry" onClick={onRetry}>
            <RefreshIcon className="w-4 h-4" />
            Tentar novamente
          </button>
        </div>
      )}
    </div>
  );
}

// Inline error for smaller spaces
export function InlineError({
  message, onRetry
}: { message: string; onRetry?: () => void }) {
  return (
    <div className="error-inline">
      <AlertCircleIcon className="error-inline-icon" />
      <span className="error-inline-message">{message}</span>
      {onRetry && (
        <span className="error-inline-action" onClick={onRetry}>
          Tentar novamente
        </span>
      )}
    </div>
  );
}

// Card with error overlay
export function MetricCard({
  title, value, trend, error, onRetry
}: MetricCardProps) {
  return (
    <div className="relative rounded-xl border p-6">
      <div style={{ filter: error ? 'blur(2px)' : 'none', opacity: error ? 0.3 : 1 }}>
        <p className="text-sm text-gray-500">{title}</p>
        <p className="text-3xl font-bold">{value}</p>
        <p className={`text-sm ${trend >= 0 ? 'text-success-500' : 'text-error-500'}`}>
          {trend >= 0 ? '+' : ''}{trend}%
        </p>
      </div>
      {error && (
        <div className="error-overlay">
          <AlertCircleIcon className="error-overlay-icon" />
          <p className="error-overlay-message">{error}</p>
          {onRetry && (
            <span className="error-overlay-retry" onClick={onRetry}>
              <RefreshIcon className="w-3 h-3" /> Atualizar
            </span>
          )}
        </div>
      )}
    </div>
  );
}

// Error toast hook
function useErrorToast() {
  const [toast, setToast] = useState<{ title: string; message: string } | null>(null);

  const showError = (title: string, message: string) => {
    setToast({ title, message });
    setTimeout(() => setToast(null), 5000);
  };

  return { toast, showError, dismiss: () => setToast(null) };
}
